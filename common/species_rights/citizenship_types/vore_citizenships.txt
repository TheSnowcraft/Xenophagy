# from = country
# this = species

citizenship_vore_fodder = {
	new_pop_weight = 0.5

	triggered_pop_group_modifier = {
		potential = {
			OR = {
				has_trait = trait_vore_prey
				has_trait = trait_vore_prey_artifical
			}
		}
		pop_happiness = +0.5
	}

	triggered_pop_group_modifier = {
		potential = {
			NOR = {
				has_trait = trait_vore_prey
				has_trait = trait_vore_prey_artifical
				has_trait = trait_vore_pred
			}
		}
		pop_happiness = -0.75
	}

	triggered_pop_group_modifier = {
		potential = {
			has_trait = trait_vore_pred
		}
		pop_happiness = -0.95
	}

	pop_group_modifier = {
		pop_amenities_usage_mult = -0.25
		pop_political_power = -0.75
	}

	potential = {
        from = {
			OR = {
            	is_pred_empire = yes
            	is_predbot_empire = yes
        	}
			NOT = { has_policy_flag = predation_policy_banned }
		}
	}

	allow = {
		custom_tooltip = {
			fail_text = citizenship_vore_fodder_requires_organic
			has_trait = trait_organic
		}
		custom_tooltip = {
			fail_text = same_species_disallowed
			NOT = { is_same_species = from }
		}
		custom_tooltip = {
			fail_text = empire_xenophile
			from = {
				is_xenophile = no
			}
		}
	}
	ai_will_do = {
		factor = 10
		modifier = {
			factor = 20
			OR = {
				has_trait = trait_vore_prey
				has_trait = trait_vore_prey_artifical
			}
		}
		modifier = {
			factor = 0.5
			NOT = {
				has_trait = trait_vore_prey
				has_trait = trait_vore_prey_artifical
			}
		}
		modifier = {
			factor = 10
			from = {
				is_fanatic_xenophobe = yes
			}
		}
		modifier = {
			factor = 5
			from = {
				is_xenophobe = yes
			}
		}
		modifier = {
			factor = 0.25
			has_trait = trait_vore_pred
		}
	}
}
